Posts

Get all post

fetch("https://json-restful-endpoint.herokuapp.com/post")
.then((response) =>  response.json())
.then((data) =>  console.log(data));

Returns 200 posts

Get fixed no. post

fetch("https://json-restful-endpoint.herokuapp.com/post?n=10")
.then((response) =>  response.json())
.then((data) =>  console.log(data));

Get single post

fetch("https://json-restful-endpoint.herokuapp.com/post/7")
.then((response) => response.json())
.then((data) => console.log(data));

Create Post

fetch("https://json-restful-endpoint.herokuapp.com/post",{
    method:  "POST",
    headers: {
    Accept:  "application/json",
    "Content-Type":  "application/json",
    },
    body:  JSON.stringify({ 
        title:  "Post title", 
        content:  "Post content" 
    }),
})
.then((response) =>  response.json())
.then((json) =>  console.log(json));

Delete Post

fetch("https://json-restful-endpoint.herokuapp.com/post/5",{
    method:  "DELETE",
    headers: {
    Accept:  "application/json",
    "Content-Type":  "application/json",
    },
})
.then((response) =>  response.json())
.then((json) =>  console.log(json));
